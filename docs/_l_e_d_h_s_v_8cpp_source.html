215 void LEDHSV::HSV2RGB(){
216  float r, g, b;
217 
218  float h = hue;
219  float s = saturation;
220  float v = brightness;
221 
222  int i = floor(h * 6);
223  float f = h * 6 - i;
224  float p = v * (1 - s);
225  float q = v * (1 - f * s);
226  float t = v * (1 - (1 - f) * s);
227 
228  switch(i % 6){
229  case 0: r = v, g = t, b = p; break;
230  case 1: r = q, g = v, b = p; break;
231  case 2: r = p, g = v, b = t; break;
232  case 3: r = p, g = q, b = v; break;
233  case 4: r = t, g = p, b = v; break;
234  case 5: r = v, g = p, b = q; break;
235  }
236 
237  setRGB(round(r*255),round(g*255),round(b*255));
238 }
